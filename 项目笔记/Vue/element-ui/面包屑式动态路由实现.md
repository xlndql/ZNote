```html
<template> 
<!-- 路由导航--> 
	<div> 
		<el-breadcrumb separator-class="el-icon-arrow-right"> 
			<el-breadcrumb-item v-for="(item,index) in breadList" :key="index" >
				{{ item.name }} 
			</el-breadcrumb-item> 
		</el-breadcrumb> 
	</div> 
</template> 
<script> 
expert default{ 
name: '', data() { return { breadList: [], // 路由集合 }; }, watch: { $route() { this.getBreadcrumb(); } }, methods: { isHome(route) { return route.name === "首页"; }, getBreadcrumb() { let matched = this.$route.matched; //如果不是首页 if (!this.isHome(matched[0])) { matched = [{path: "/home", name: '首页'}].concat(matched); } this.breadList = matched; }, } } </script>
```
